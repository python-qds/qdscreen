23 def df_mix1():
24 """The dataset for reuse"""
25 df = pd.DataFrame({
26 'U': ["a", "b", "d", "a", "b", "c", "a", "b", "d", "c"],
27 'V': ["a", "b", "c", "a", "b", "c", "a", "b", "c", "c"],
28 'W': ["a", "b", "b", "a", "b", "b", "a", "b", "b", "b"],
29 'X': ["a", "a", "b", "b", "a", "c", "c", "a", "b", "c"],
30 'Y': ["b", "b", "c", "c", "b", "a", "a", "b", "c", "a"],
31 'Z': ["a", "a", "b", "a", "a", "b", "b", "a", "a", "b"] # if the 3d element were an "a", X->Z and Y->Z
32 })
33  
34 # In this data: the following relationships hold:
35 # H(W|V) = 0 H(W|U) = 0 H(V|U) = 0 # U -> V -> W
36 # H(Y|X) = H(X|Y) = 0 # Y <-> X but X selected as representant as 1st appearing
37 # H(Z|Y) = H(Z|X) = 0.275 AND H(Z|Y) / H(Z) = 0.284 # X -> Z if qd threshold > 0.19 (absolute) or 0.28 (relative)
38 return df
